Automação de Análise de Dados

Script Python para automatização de análise de dados em planilhas Excel, com geração de insights e relatórios automáticos.

Funcionalidades

  • Leitura automática de planilhas Excel
  • Cálculo de métricas importantes (total, média, tendências)
  • Análise de tendências com regressão linear
  • Geração de relatórios automáticos

Tecnologias Utilizadas

Python 3.9
Pandas
NumPy
Openpyxl

Código do Projeto


import pandas as pd
import numpy as np
from datetime import datetime

class AnalisadorDados:
    def __init__(self, arquivo):
        self.df = pd.read_excel(arquivo)
        
    def analise_basica(self):
        return {
            'total_vendas': self.df['vendas'].sum(),
            'media_vendas': self.df['vendas'].mean(),
            'max_venda': self.df['vendas'].max(),
            'min_venda': self.df['vendas'].min()
        }
    
    def analise_tendencia(self):
        x = np.arange(len(self.df))
        y = self.df['vendas'].values
        
        # Regressão linear
        coef = np.polyfit(x, y, 1)
        tendencia = coef[0]  # Coeficiente angular
        
        return {
            'tendencia': tendencia,
            'crescimento': tendencia > 0
        }
    
    def gerar_relatorio(self):
        analise = self.analise_basica()
        tendencia = self.analise_tendencia()
        
        with open(f'relatorio_{datetime.now():%Y%m%d}.txt', 'w') as f:
            f.write("=== Relatório de Análise ===\n\n")
            f.write(f"Total de Vendas: R$ {analise['total_vendas']:,.2f}\n")
            f.write(f"Média de Vendas: R$ {analise['media_vendas']:,.2f}\n")
            f.write(f"Maior Venda: R$ {analise['max_venda']:,.2f}\n")
            f.write(f"Menor Venda: R$ {analise['min_venda']:,.2f}\n\n")
            f.write("=== Análise de Tendência ===\n")
            f.write(f"Tendência: {'Crescimento' if tendencia['crescimento'] else 'Queda'}\n")
            f.write(f"Taxa: {tendencia['tendencia']:.2%} por período\n")

if __name__ == '__main__':
    analisador = AnalisadorDados('dados_vendas.xlsx')
    analisador.gerar_relatorio()
